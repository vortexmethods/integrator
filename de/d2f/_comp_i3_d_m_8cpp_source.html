28 double CompI3DM::evaluate(int i, int j)
29 {
30  //Точка наблюдения
31  const v3D& pt = db.point[i];
32 
33  //Векторы из вершин панели в точку наблюдения
34  const v3D& va = pt - db.node[db.topo[j][0]];
35  const v3D& vb = pt - db.node[db.topo[j][1]];
36  const v3D& vc = pt - db.node[db.topo[j][2]];
37 
38  v3D ova(va), ovb(vb), ovc(vc);
39 
40  //
41  double lva = ova.normalize();
42  double lvb = ovb.normalize();
43  double lvc = ovc.normalize();
44 
45  //Орты векторов, направленных вдоль сторон треугольной панели K_j, лежащих против соответствующих вершин
46  v3D taua = db.node[db.topo[j][2]] - db.node[db.topo[j][1]];
47  v3D taub = db.node[db.topo[j][0]] - db.node[db.topo[j][2]];
48  v3D tauc = db.node[db.topo[j][1]] - db.node[db.topo[j][0]];
49 
50  //Длины сторон треугольника, лежащие против соответствующих вершин
51  double La = taua.normalize();
52  double Lb = taub.normalize();
53  double Lc = tauc.normalize();
54 
55  //phi^p_q - плоский угол при основании боковой грани получающегося тетраэдра
56  //(между вектором v_p и стороной L_q)
57  double fca = ova & tauc;
58  double fcb = -(ovb & tauc);
59  double fab = ovb & taua;
60  double fac = -(ovc & taua);
61  double fbc = ovc & taub;
62  double fba = -(ova & taub);
63 
64  v3D Phi = (va ^ vb) * (log((lva * (1.0 + fca)) / (lvb * (1.0 - fcb))) / Lc)
65  + (vb ^ vc) * (log((lvb * (1.0 + fab)) / (lvc * (1.0 - fac))) / La)
66  + (vc ^ va) * (log((lvc * (1.0 + fbc)) / (lva * (1.0 - fba))) / Lb);
67 
68  //Величина, равная по модулю телесному углу, под которым видна панель из точки наблюдения
69  double Theta = 2.0 * atan2((ova ^ ovb) & ovc, 1.0 + (ova & ovb) + (ovb & ovc) + (ovc & ova));
70 
71  return 0.25/M_PI * ((Phi - va * Theta) & db.nrm[j]);
72 }
