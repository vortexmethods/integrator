55 v3D CompI3DK_Duffy::DistContact(double xi1, double xi2, double eta1, double eta2, const i3D& nodes_i, const i3D& nodes_j)
56 {
57  const v3D& P11 = db.node[nodes_i[0]];
58  const v3D& P12 = db.node[nodes_i[1]];
59  const v3D& P13 = db.node[nodes_i[2]];
60  const v3D& P22 = db.node[nodes_j[1]];
61  const v3D& P23 = db.node[nodes_j[2]];
62 
63  return (eta1 - xi1) * P11 + (xi1 - xi2) * P12 + xi2 * P13 - (eta1 - eta2) * P22 - eta2 * P23;
64 }

66 v3D CompI3DK_Duffy::DistSosed(double xi2, double u1, double u2, const i3D& nodes_i, const i3D& nodes_j)
67 {

72  const v3D& P23 = db.node[nodes_j[2]];
73

121  [&](const v2D& r) {return r[0] / DistSosed(1.0-r[0], r[0], r[0]*r[1], nodes_i, nodes_j).length(); },

126  };

157 double CompI3DK_Duffy::evaluate(int i, int j)
158 {
159  size_t locRefine = nrefine;
160 
161  if (i==j)
162  //if (&(db.topo[i]) == &(db.topo[j]))
163  {
164  return 0.0;
165  }
166  if (db.ifSosed({ i, j }))
167  {
168  i2D shifts = RenumerationSosed(i, j);
169  auto Integral = IntDuffySosed({ i, shifts[0] }, { j, shifts[1] }, locRefine);
170  //if (locRefine == maxRefine)
171  // std::cout << "Sosed, (i,j) = " << i2D{ i,j } << std::endl;
172  //std::cout << "locRefine = " << locRefine << std::endl;
173  return Integral;
174  }
175 
176  if (db.ifContact({ i, j }))
177  {
178  i2D shifts = RenumerationContact(i, j);
179  auto Integral = IntDuffyContact({ i, shifts[0] }, { j, shifts[1] }, locRefine);
180  //if (locRefine == maxRefine)

182  return Integral;

186  // Дальние панели
187  auto onePt = [i, j, this](v3D pt)
188  {
189  //Векторы из вершин панели в точку наблюдения
190  const v3D& va = pt - db.node[db.topo[j][0]];
191  const v3D& vb = pt - db.node[db.topo[j][1]];
192  const v3D& vc = pt - db.node[db.topo[j][2]];
193 
194  v3D ova(va), ovb(vb), ovc(vc);
195 
196  //
197  double lva = ova.normalize();
198  double lvb = ovb.normalize();
199  double lvc = ovc.normalize();
200 
201  //Орты векторов, направленных вдоль сторон треугольной панели K_j, лежащих против соответствующих вершин
202  v3D taua = db.node[db.topo[j][2]] - db.node[db.topo[j][1]];
203  v3D taub = db.node[db.topo[j][0]] - db.node[db.topo[j][2]];
204  v3D tauc = db.node[db.topo[j][1]] - db.node[db.topo[j][0]];
205 
206  //Длины сторон треугольника, лежащие против соответствующих вершин
207  double La = taua.normalize();
208  double Lb = taub.normalize();
209  double Lc = tauc.normalize();
210 
211  //phi^p_q - плоский угол при основании боковой грани получающегося тетраэдра
212  //(между вектором v_p и стороной L_q)
213  double fca = ova & tauc;
214  double fcb = -(ovb & tauc);
215  double fab = ovb & taua;
216  double fac = -(ovc & taua);
217  double fbc = ovc & taub;
218  double fba = -(ova & taub);
219 
220  v3D Phi = (va ^ vb) * (log((lva * (1.0 + fca)) / (lvb * (1.0 - fcb))) / Lc)
221  + (vb ^ vc) * (log((lvb * (1.0 + fab)) / (lvc * (1.0 - fac))) / La)
222  + (vc ^ va) * (log((lvc * (1.0 + fbc)) / (lva * (1.0 - fba))) / Lb);
223 
224  //Величина, равная по модулю телесному углу, под которым видна панель из точки наблюдения
225  double Theta = 2.0 * atan2((ova ^ ovb) & ovc, 1.0 + (ova & ovb) + (ovb & ovc) + (ovc & ova));
226 
227  return 0.25 / M_PI * ((Phi - va * Theta) & db.nrm[j]);
228  };
