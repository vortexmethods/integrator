40 p13D CompJ3DK::ThetaPsi(const v3D& pt, int j)
41 {
42  //Векторы из вершин панели в точку наблюдения
43  const v3D& va = pt - db.node[db.topo[j][0]];
44  const v3D& vb = pt - db.node[db.topo[j][1]];
45  const v3D& vc = pt - db.node[db.topo[j][2]];
46 
47  v3D ova(va), ovb(vb), ovc(vc);
48 
49  double lva = ova.normalize();
50  double lvb = ovb.normalize();
51  double lvc = ovc.normalize();
52 
53  //Орты векторов, направленных вдоль сторон треугольной панели K_j, лежащих против соответствующих вершин
54  v3D taua = db.node[db.topo[j][2]] - db.node[db.topo[j][1]];
55  v3D taub = db.node[db.topo[j][0]] - db.node[db.topo[j][2]];
56  v3D tauc = db.node[db.topo[j][1]] - db.node[db.topo[j][0]];
57 
58  taua.normalize();
59  taub.normalize();
60  tauc.normalize();
61 
62  double rac = (ova & tauc), rbc = (ovb & tauc), rba = (ovb & taua), rca = (ovc & taua), rcb = (ovc & taub), rab = (ova & taub);
63 
64  //auto w1 = ova ^ ovb;
65  //auto w2 = (ova ^ ovb) & ovc;
66  //auto w3 = 1.0 + (ova & ovb) + (ovb & ovc) + (ovc & ova);
67 
68  double term1, term2, term3;
69 
70  //Проверка частных случаев
71  //if ((fabs(rac + 1.0) < 0.5*eps_psi_theta2) && (fabs(rbc + 1.0) < 0.5*eps_psi_theta2))
72  if (fabs(rbc + 1.0) < 0.5 * eps_psi_theta2)
73  term1 = log(lvb / lva);
74  else
75  term1 = log((lva * (1.0 + (ova & tauc))) / (lvb * (1.0 + (ovb & tauc))));
76 
77  //if ((fabs(rba + 1.0) < 0.5 * eps_psi_theta2) && (fabs(rca + 1.0) < 0.5 * eps_psi_theta2))
78  if (fabs(rca + 1.0) < 0.5 * eps_psi_theta2)
79  term2 = log(lvc / lvb);
80  else
81  term2 = log((lvb * (1.0 + (ovb & taua))) / (lvc * (1.0 + (ovc & taua))));
82 
83  //if ((fabs(rcb + 1.0) < 0.5 * eps_psi_theta2) && (fabs(rab + 1.0) < 0.5 * eps_psi_theta2))
84  if (fabs(rab + 1.0) < 0.5 * eps_psi_theta2)
85  term3 = log(lva / lvc);
86  else
87  term3 = log((lvc * (1.0 + (ovc & taub))) / (lva * (1.0 + (ova & taub))));
88 
89  const v3D& Psi = term1 * tauc + term2 * taua + term3 * taub;
90 
91  //Величина, равная по модулю телесному углу, под которым видна панель из точки наблюдения
92  double Theta = 2.0 * atan2((ova ^ ovb) & ovc, 1.0 + (ova & ovb) + (ovb & ovc) + (ovc & ova));
93 
94  return { Theta, Psi };
95 }

153 p13D CompJ3DK::SingContact(const v3D& M, const i2D& ii, const i2D& jj)
154 {
155  const i3D& nodes_i = db.topo[ii[0]].rotateLeft(ii[1]);
156  const i3D& nodes_j = db.topo[jj[0]].rotateLeft(jj[1]);
157 
158  //Векторы из вершин панели в точку наблюдения
159  v3D ovc = M - db.node[nodes_j[0]];
160  double lvc = ovc.normalize();
161 
162  //Орты векторов, направленных вдоль сторон треугольной панели K_j, лежащих против соответствующих вершин
163  v3D taua = (db.node[nodes_j[0]] - db.node[nodes_j[2]]).unit();
164  v3D taub = (db.node[nodes_j[1]] - db.node[nodes_j[0]]).unit();
165 
166  v3D e = (db.nrm[ii[0]] ^ db.nrm[jj[0]]);
167  if (e.length2() < eps_zero2)
168  e = taub;
169  else
170  e.normalize();
171 
172  const v3D& nrmi = db.nrm[ii[0]];
173  const v3D& nrmj = db.nrm[jj[0]];
174 
175  double deltaA = atan2((-e ^ taua) & nrmj, -e & taua);
176  double deltaB = atan2((e ^ taub) & nrmj, e & taub);
177 
178  if (((M_PI - fabs(deltaA)) < eps_zero) || ((M_PI - fabs(deltaB)) < eps_zero))
179  {
180  e *= -1.0;
181  deltaA = atan2((-e ^ taua) & nrmj, -e & taua);
182  deltaB = atan2((e ^ taub) & nrmj, e & taub);
183  }
184 
185  if ((deltaA * deltaB < eps_zero) && (fabs(deltaA - deltaB) > M_PI))
186  {
187  e *= -1.0;
188  deltaA = atan2((-e ^ taua) & nrmj, -e & taua);
189  deltaB = atan2((e ^ taub) & nrmj, e & taub);
190  }
191 
192  double ari = sqrt(db.measure[ii[0]]);
193 
194  return { 2.0*(atan2((ovc ^ taua) & e, (e - ovc) & (e - taua)) + atan2( (ovc ^ taub) & e, (e - ovc) & (e + taub) )),
195  -(taua * log((lvc * (1 + (taua & ovc))) / ari) + taub * log((lvc * (1 - (taub & ovc))) / ari)) };
196 }

228 
229  double alpha = angle(db.node[nodes_i[0]] - db.node[nodes_i[2]], db.node[nodes_i[1]] - db.node[nodes_i[2]]);
230  double beta = angle(db.node[nodes_i[2]] - db.node[nodes_i[1]], db.node[nodes_i[0]] - db.node[nodes_i[1]]);
231  double gamma = angle(db.node[nodes_j[2]] - db.node[nodes_j[1]], db.node[nodes_j[0]] - db.node[nodes_j[1]]);
232  double delta = angle(db.node[nodes_j[1]] - db.node[nodes_j[2]], db.node[nodes_j[0]] - db.node[nodes_j[2]]);
233

236  const v3D& nrmi = db.nrm[ii[0]];
237  const v3D& nrmj = db.nrm[jj[0]];
238 
239  double xi = atan2((nrmi ^ nrmj) & tauc, nrmi & nrmj);

241  double sinXi = sin(xi), cosXi = cos(xi);
242 
243  // И.К. Добавил предвычисление синусов-косинусов
244  double sinAlpha = sin(alpha), cosAlpha = cos(alpha);
245  double sinBeta = sin(beta), cosBeta = cos(beta);
246 
247  // И.К. Вычислим их здесь, чтобы однократно
248  double sinNu = sinAlpha * cosBeta + cosAlpha * sinBeta, cosNu = sinAlpha * sinBeta - cosAlpha * cosBeta;
249 
250  double sinGamma = sin(gamma), cosGamma = cos(gamma);
251  double sinDelta = sin(delta), cosDelta = cos(delta);
252 
253 
254  double cosSigma = -(cosAlpha * cosDelta + cosXi * sinAlpha * sinDelta);
255  double cosMu = -(cosBeta * cosGamma + cosXi * sinBeta * sinGamma);
256  double cosLambda = -(cosAlpha * cosGamma - cosXi * sinAlpha * sinGamma);
257  double cosTheta = -(cosBeta * cosDelta - cosXi * sinBeta * sinDelta);
258 
259  double q_alpha_beta = sinNu * log(tan(alpha / 2) * tan(nu / 2)) / sin(beta) + sinNu * log(tan(beta / 2) * tan(nu / 2)) / sin(alpha) \
260  + log(tan(alpha / 2) * tan(beta / 2));
261 
262  auto phi = [sinXi](double sinAlpha, double cosAlpha, double sinGamma, double cosGamma, double cosLambda)
263  {
264  //std::cout << sinXi * sinAlpha * sinGamma / (1.0 - cosAlpha + cosGamma + cosLambda) << std::endl;
265  return 2.0*atan2(sinXi * sinAlpha * sinGamma, 1.0 - cosAlpha + cosGamma + cosLambda);

289  sinNu * (cosBeta + cosMu * cosGamma) * log(1.0 - cosGamma) + \
290  sinBeta * (1.0 - cosMu) * (cosNu - cosLambda) * log(sinBeta / sinNu) + \
291  sinNu * sinBeta * (sinBeta * cosGamma - cosXi * sinGamma * cosBeta) * log((1.0 - cosNu) / (1.0 + cosBeta)) + \
292  phi2 * sinXi * sinGamma * sinNu * sinBeta )
293  };
294  }; // []q_Theta_Psi(...)
