25 double CompI3DK::evaluate(int i, int j)
26 {
27  auto onePt = [i, j, this](v3D pt)
28  {
29  //Векторы из вершин панели в точку наблюдения
30  const v3D& va = pt - db.node[db.topo[j][0]];
31  const v3D& vb = pt - db.node[db.topo[j][1]];
32  const v3D& vc = pt - db.node[db.topo[j][2]];
33 
34  v3D ova(va), ovb(vb), ovc(vc);
35 
36  //
37  double lva = ova.normalize();
38  double lvb = ovb.normalize();
39  double lvc = ovc.normalize();
40 
41  //Орты векторов, направленных вдоль сторон треугольной панели K_j, лежащих против соответствующих вершин
42  v3D taua = db.node[db.topo[j][2]] - db.node[db.topo[j][1]];
43  v3D taub = db.node[db.topo[j][0]] - db.node[db.topo[j][2]];
44  v3D tauc = db.node[db.topo[j][1]] - db.node[db.topo[j][0]];
45 
46  //Длины сторон треугольника, лежащие против соответствующих вершин
47  double La = taua.normalize();
48  double Lb = taub.normalize();
49  double Lc = tauc.normalize();
50 
51  //phi^p_q - плоский угол при основании боковой грани получающегося тетраэдра
52  //(между вектором v_p и стороной L_q)
53  double fca = ova & tauc;
54  double fcb = -(ovb & tauc);
55  double fab = ovb & taua;
56  double fac = -(ovc & taua);
57  double fbc = ovc & taub;
58  double fba = -(ova & taub);
59 
60  v3D Phi = (va ^ vb) * (log((lva * (1.0 + fca)) / (lvb * (1.0 - fcb))) / Lc)
61  + (vb ^ vc) * (log((lvb * (1.0 + fab)) / (lvc * (1.0 - fac))) / La)
62  + (vc ^ va) * (log((lvc * (1.0 + fbc)) / (lva * (1.0 - fba))) / Lb);
63 
64  //Величина, равная по модулю телесному углу, под которым видна панель из точки наблюдения
65  double Theta = 2.0 * atan2((ova ^ ovb) & ovc, 1.0 + (ova & ovb) + (ovb & ovc) + (ovc & ova));
66 
67  return 0.25 / M_PI * ((Phi - va * Theta) & db.nrm[j]);
68  };
