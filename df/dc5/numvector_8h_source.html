869  auto angle(const numvector<T, n>& x, const numvector<P, n>& y) -> typename std::remove_const<decltype(x[0] * y[0])>::type
870  {
871  auto den = sqrt((x.length2() * y.length2()));
872  if (den == 0)
873  return 0;
874 
875  typename std::remove_const<decltype(x[0] * y[0])>::type res = (x & y) / den;
876  if (res >= 1.0)
877  return 0;
878  if (res <= -1.0)
879  return (typename std::remove_const<decltype(x[0] * y[0])>::type)3.1415926535897932;
880 
881  return acos(res);
882  }
