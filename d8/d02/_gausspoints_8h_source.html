284 template<size_t dim>
285 Gausspoints<dim>::Gausspoints(const Database<dim>& db_, const gpPosWeight<dim>& pw_)
286  : ngp(pw_.weight.size()), db(db_), pw(pw_)
287 {
288  Lcoord.resize(ngp);
289 
290  for (size_t i = 0; i < ngp; ++i)
291  {
292  for (size_t s = 0; s < dim - 1; ++s)
293  Lcoord[i][s] = pw.pos[i][s];
294 
295  double lastLcoord = 1.0;
296  for (size_t s = 0; s < dim - 1; ++s)
297  lastLcoord -= Lcoord[i][s];
298  Lcoord[i][dim - 1] = lastLcoord;
299  }
300 };//Gausspoints(...)

303 template<size_t dim> template<typename T>
304 T Gausspoints<dim>::integrate(std::function<T(const numvector<double, dim>&)> fo, size_t pnl, size_t refineLevel) const
305 {
306  const auto& topo = db.topo[pnl];
307 
308  simplex nds;
309  for (size_t q = 0; q < dim; ++q)
310  nds.first[q] = db.node[topo[q]];
311  nds.second = db.measure[pnl];
312 
313  std::vector trgs { nds };
314  for (int r = 0; r < refineLevel; ++r)
315  refine(trgs);
316 
317  numvector<double, dim> pt;
318  T res(0.0);
319 
320  for (auto& tr : trgs)
321  {
322  for (size_t i = 0; i < ngp; ++i)
323  {
324  pt.toZero();
325  for (size_t s = 0; s < dim; ++s)
326  pt += tr.first[s] * Lcoord[i][s];
327 
328  res += (tr.second * pw.weight[i]) * fo(pt);
329  //std::cout << "pt = " << pt << ", f(pt) = " << fo(pt) << std::endl;
330  }//for i
331  }
332  return res;
333 }//integrate(...)
