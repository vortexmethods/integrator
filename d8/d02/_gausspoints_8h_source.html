449 template<size_t dim> template<typename T>
450 T Gausspoints<dim>::integrate(std::function<T(const numvector<double, dim>&)> fo, size_t pnl, size_t refineLevel) const
451 {
452  const auto& topo = db.topo[pnl];
453 
454  simplex nds;
455  for (size_t q = 0; q < dim; ++q)
456  nds.first[q] = db.node[topo[q]];
457  nds.second = db.measure[pnl];
458 
459  std::vector trgs { nds };
460  for (int r = 0; r < refineLevel; ++r)
461  refine(trgs);
462 
463  numvector<double, dim> pt;
464  T res(0.0);
465 
466  for (auto& tr : trgs)
467  {
468  for (size_t i = 0; i < ngp; ++i)
469  {
470  pt.toZero();
471  for (size_t s = 0; s < dim; ++s)
472  pt += tr.first[s] * Lcoord[i][s];
473 
474  res += (tr.second * pw.weight[i]) * fo(pt);
475  //std::cout << "pt = " << pt << ", f(pt) = " << fo(pt) << std::endl;
476  }//for i
477  }
478  return res;
479 }//integrate(...)

626 {
627 
628 
629  numvector<double, dim> zeroVec;
630  zeroVec.toZero();
631 
632  std::vector<std::pair<numvector<double, dim>, double>> cubs{ { zeroVec, 1.0 } };
633  for (int r = 0; r < refineLevel; ++r)
634  refineCube(cubs);
635 
636 
637  numvector<T, dim> resijk;
638  numvector<double, dim> posijk;
639  numvector<double, dim> weightijk;
640 
641  T result(0);
642 
643  double h;
644 
645  if (dim == 2)
646  {
647  for (auto& cb : cubs)
648  {
649  h = cb.second;
650  const numvector<double, dim>& a = cb.first;
651 
652 
653  resijk[0] = 0.0;
654  for (size_t i = 0; i < pw.pos.size(); ++i)
655  {
656  posijk[0] = a[0] + h * pw.pos[i][0];
657  weightijk[0] = pw.weight[i];
658 
659  resijk[1] = 0.0;
660  for (size_t j = 0; j < pw.pos.size(); ++j)
661  {
662  posijk[1] = a[1] + h * pw.pos[j][0];
663  weightijk[1] = pw.weight[j];
664 
665  resijk[1] += weightijk[1] * fo(posijk);
666  }//for j
667 
668  resijk[0] += weightijk[0] * resijk[1];
669  }//for i
670 
671  result += resijk[0];
672  }
673 
674  result *= h * h;
675  }
676 
677  if (dim == 3)
678  {
679  for (auto& cb : cubs)
680  {
681  h = cb.second;
682  const numvector<double, dim>& a = cb.first;
683 
684  resijk[0] = 0.0;
685  for (size_t i = 0; i < pw.pos.size(); ++i)
686  {
687  posijk[0] = a[0] + h * pw.pos[i][0];
688  weightijk[0] = pw.weight[i];
689 
690  resijk[1] = 0.0;
691  for (size_t j = 0; j < pw.pos.size(); ++j)
692  {
693  posijk[1] = a[1] + h * pw.pos[j][0];
694  weightijk[1] = pw.weight[j];
695 
696  resijk[2] = 0.0;
697  for (size_t k = 0; k < pw.pos.size(); ++k)
698  {
699  posijk[2] = a[2] + h * pw.pos[k][0];
700  weightijk[2] = pw.weight[k];
701 
702  resijk[2] += weightijk[2] * fo(posijk);
703  }//for k
704 
705  resijk[1] += weightijk[1] * resijk[2];
706  }//for j
707 
708  resijk[0] += weightijk[0] * resijk[1];
709  }//for i
