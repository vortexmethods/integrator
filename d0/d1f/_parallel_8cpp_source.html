90 parProp Parallel::SplitMPI(size_t n, bool bcastAll) const
91 {
92  parProp par;
93 
94  par.comm = comm;
95  par.np = np;
96  par.rank = rank;
97 
98  par.totalLen = static_cast<int>(n);
99  MPI_Bcast(&par.totalLen, 1, MPI_INT, 0, comm);
100 
101  if (rank == 0)
102  {
103  par.len.clear();
104  par.disp.clear();
105 
106  int nPerP = static_cast<int>(n / np);
107 
108  for (int s = 0; s < np - 1; ++s)
109  {
110  par.len.push_back(nPerP);
111  par.disp.push_back(s*nPerP);
112  }
113 
114  par.len.push_back(static_cast<int>(n) - nPerP * (np - 1));
115  par.disp.push_back(nPerP * (np - 1));
116  }
117 
118  MPI_Scatter(par.len.data(), 1, MPI_INT, &par.myLen, 1, MPI_INT, 0, comm);
119  MPI_Scatter(par.disp.data(), 1, MPI_INT, &par.myDisp, 1, MPI_INT, 0, comm);
120 
121  if (bcastAll)
122  {
123  if (rank != 0)
124  {
125  par.len.resize(np);
126  par.disp.resize(np);
127  }
128  MPI_Bcast(par.len.data(), np, MPI_INT, 0, comm);
129  MPI_Bcast(par.disp.data(), np, MPI_INT, 0, comm);
130  }
131 
132  return par;
133 
134 }//SplitMPI(...)
